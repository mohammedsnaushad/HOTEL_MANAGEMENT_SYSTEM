CREATE PROCEDURE [dbo].[twocolumnsdata] @UserName varchar(50),@Password varchar(50)
AS
BEGIN
SELECT count(UserName) as matches FROM [dbo].[UserRegistration]
WHERE UserName=@UserName AND Password=@Password
END
GO
